    SELECTION GAIN¶

    Seleksi Fitur sangat berguna dalam Data Mining karena dapat menghemat waktu, biaya, dan tenaga dengan cara membuang fitur yang tidak penting. jika kita menganalisa banyak kolom dengan kita menyeleksi fitur data tersebut kita dapat mengurangi kolom data itu sehingga membuat waktu analisis lebih cepat.

import math
def entropy(x,y):
    hasil = -x/14*math.log(x/14,2)-y/14*math.log(y/14,2)
    return "ENTROPY TARGET :",hasil

import math
def overcast(y,n):
    yes=y/4
    no=n/4
    data=(yes+no)/14
    entro=-yes/14*math.log(yes/14,2)
    hasil = data*entro
    print ("overcast_yes         :",yes)
    print ("overcast_no          :",no)
    print ("data overcast yes+no :",data)
    print ("entro:",entro)
    return (hasil)
def sunny(y,n):
    yes=y/5
    no=n/5
    data=(yes+no)/14
    entro=-yes/14*math.log(yes/14,2)-no/14*math.log(no/14,2)
    hasil = data*entro

    print ("sunnyt_no          :",no)
